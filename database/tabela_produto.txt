CREATE TABLE produto (
idproduto INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
nome VARCHAR(50) NOT NULL,
descricao VARCHAR (100) NOT NULL,
quantidade VARCHAR (15) NOT NULL,
imagem VARCHAR(50) DEFAULT 'foto.png' NOT NULL,
reg_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
idcategoria INT NOT NULL,
CONSTRAINT pk_categoria FOREIGN KEY (idcategoria) REFERENCES categorias(codcategoria))ENGINE=InnoDB;
